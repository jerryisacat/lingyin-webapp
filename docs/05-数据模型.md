## 1. Prisma Schema

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"   // Supabase PostgreSQL
  url      = env("DATABASE_URL")
}

// ─── 用户 ─────────────────────────────────────────

model User {
  id        String   @id    @default(cuid())
  email     String   @unique
  nickname  String?
  tone      String   @default("warm")  // 上次选择的语气

  // 订阅 (Phase 3)
  subscription     String?  @default("free") // "free" | "basic" | "pro"
  subscriptionId   String?          // Stripe subscription ID
  subscribedAt     DateTime?
  subscriptionEnds DateTime?
  dailyEntryCount  Int      @default(0)    // 当天已写日记数
  dailyEntryDate   DateTime?               // 计数日期

  // 统计
  totalEntries    Int      @default(0)
  storageBytes    BigInt   @default(0)     // 已用存储 (字节)
  lastEntryAt     DateTime?
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  entries Entry[]
}

// ─── 日记元数据 ────────────────────────────────────
// 注意：日记正文（Markdown）存储在 CloudFlare R2，
// 数据库只存元数据用于查询

model Entry {
  id         String   @id    @default(cuid())
  userId     String
  date       DateTime        // 日记日期 (YYYY-MM-DD)
  title      String?
  tone       String   @default("warm")

  // 元数据
  wordCount    Int      @default(0)
  preview      String?          // Markdown 前 200 字，用于时间线列表
  hasImages    Boolean  @default(false)
  imageCount   Int      @default(0)
  thumbnailUrl String?

  // 存储信息 (R2)
  markdownPath String               // R2 路径 (例如 "users/{userId}/entries/2026/05/2026-05-19.md")

  // 标签
  tags        String?               // JSON array: ["旅行", "美食"]
  location    String?

  // 时间
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, date])    // 每人每天最多一篇日记
  @@index([userId, date])
  @@index([userId, createdAt])
}

// ─── (Phase 3) 订阅记录 ──────────────────────────

model Subscription {
  id             String   @id @default(cuid())
  userId         String   @unique
  plan           String   // "basic" | "pro"
  status         String   // "active" | "canceled" | "past_due"
  stripeId       String?  // Stripe subscription ID
  stripePriceId  String?  // Stripe price ID
  currentPeriodStart DateTime?
  currentPeriodEnd   DateTime?
  canceledAt     DateTime?
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt
}

// ─── (Phase 3) 账单历史 ──────────────────────────

model Invoice {
  id             String   @id @default(cuid())
  userId         String
  stripeInvoiceId String? // Stripe invoice ID
  amount         Int      // 金额 (分)
  currency       String   @default("cny")
  status         String   // "paid" | "unpaid"
  paidAt         DateTime?
  createdAt      DateTime @default(now())

  @@index([userId])
}
```

## 2. 文件存储结构 (CloudFlare R2)

所有文件使用 **CloudFlare R2**（S3 兼容），通过 `@aws-sdk/client-s3` 读写，按用户和日期组织：

```
{bucket}/users/{userId}/entries/2026/05/
  ├── 2026-05-19.md          ← 日记正文 (Markdown)
  └── assets/                ← 图片
      ├── IMG_20260519_001.jpg
      └── IMG_20260519_002.png
```

> **注意:** Vercel 无持久化文件系统，开发环境 `data/` 仅供本地模拟，生产必须走 R2。

## 4. TypeScript 类型定义

```typescript
// src/types/index.ts

// ─── 用户配置 ─────────────────────────────
export interface UserConfig {
  apiProvider: 'openai' | 'deepseek' | 'gemini';
  tone: Tone;
  // 注意: apiKey 不在此对象中，仅存 localStorage，不在数据库
}

// ─── 日记 ─────────────────────────────
export interface DiaryEntry {
  id: string;
  userId: string;
  date: string;           // "2026-05-19"
  title?: string;
  markdown: string;       // 完整 Markdown 内容 (从 R2 读取)
  tone: Tone;
  tags: string[];
  images: MediaFile[];
  wordCount: number;
  createdAt: string;
  updatedAt: string;
}

export interface DiarySummary {
  id: string;
  date: string;
  title?: string;
  preview: string;        // Markdown 前 200 字 (来自 Entry.preview)
  hasImages: boolean;
  wordCount: number;
  tags: string[];
  createdAt: string;
}

export interface MediaFile {
  url: string;          // R2 公开访问 URL
  path: string;         // R2 存储路径
  type: 'image' | 'video';  // Phase 1 仅 image
  mime: string;         // "image/jpeg" | "image/png" | "image/webp"
  size: number;         // bytes
  width?: number;
  height?: number;
  thumbnail?: string;   // 缩略图 URL
}

// ─── AI ───────────────────────────────
export type Tone = 'warm' | 'genki' | 'minimal' | 'literary';

export interface AIGenerateRequest {
  text: string;
  images: File[];
  tone: Tone;
  date: string;
}

export interface AIGenerateResponse {
  markdown: string;
  title?: string;
  tags: string[];
}

// ─── API ──────────────────────────────
export interface ApiResponse<T> {
  ok: boolean;
  data?: T;
  error?: string;
}

// ─── 表单 / 本地存储 ────────────────────
export interface LocalApiKeyStore {
  provider: 'openai' | 'deepseek' | 'gemini';
  apiKey: string;         // 仅存 localStorage，永不发给前端之外的任何地方
}
```
