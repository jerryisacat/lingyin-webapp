## 1. Prisma Schema

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"   // Supabase PostgreSQL
  url      = env("DATABASE_URL")
}

// ─── 用户 ─────────────────────────────────────────

model User {
  id            String   @id @default(cuid())
  email         String   @unique
  emailVerified DateTime?
  nickname      String?
  passwordHash  String       // bcrypt 哈希
  image         String?
  tone          String   @default("warm")

  // 订阅 (Phase 3)
  subscription     String?  @default("free")
  subscriptionId   String?
  subscribedAt     DateTime?
  subscriptionEnds DateTime?

  // 统计
  dailyEntryCount Int      @default(0)
  dailyEntryDate  DateTime?
  totalEntries    Int      @default(0)
  storageBytes    BigInt   @default(0)
  lastEntryAt     DateTime?

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  entries             Entry[]
  apiKeys             ApiKey[]
  verificationTokens  VerificationToken[]
  passwordResetTokens PasswordResetToken[]
}

// ─── 邮箱验证令牌 ────────────────────────────────

model VerificationToken {
  id        String   @id @default(cuid())
  userId    String
  token     String   @unique
  expiresAt DateTime

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([token])
}

// ─── 密码重置令牌 ────────────────────────────────

model PasswordResetToken {
  id        String   @id @default(cuid())
  userId    String
  token     String   @unique
  expiresAt DateTime

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([token])
}

// ─── API Key (服务端加密存储) ────────────────────

model ApiKey {
  id           String  @id @default(cuid())
  userId       String
  provider     String     // "openai" | "deepseek" | "gemini"
  encryptedKey String     // AES-256-GCM 加密后的 base64
  label        String?
  isActive     Boolean @default(true)
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, provider])
  @@index([userId])
}

// ─── 日记元数据 ────────────────────────────────────
// 日记正文（Markdown）存储在 CloudFlare R2，数据库只存元数据

model Entry {
  id         String   @id @default(cuid())
  userId     String
  date       DateTime
  title      String?
  tone       String   @default("warm")

  wordCount    Int      @default(0)
  preview      String?      // Markdown 前 200 字，用于时间线列表
  hasImages    Boolean  @default(false)
  imageCount   Int      @default(0)
  thumbnailUrl String?
  markdownPath String       // R2 路径

  tags     String?
  location String?

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, date])
  @@index([userId, date])
  @@index([userId, createdAt])
}

// ─── (Phase 3) 订阅记录 ──────────────────────────

model Subscription {
  id             String   @id @default(cuid())
  userId         String   @unique
  plan           String
  status         String
  stripeId       String?
  stripePriceId  String?
  currentPeriodStart DateTime?
  currentPeriodEnd   DateTime?
  canceledAt     DateTime?
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt
}

// ─── (Phase 3) 账单历史 ──────────────────────────

model Invoice {
  id             String   @id @default(cuid())
  userId         String
  stripeInvoiceId String?
  amount         Int
  currency       String   @default("cny")
  status         String
  paidAt         DateTime?
  createdAt      DateTime @default(now())

  @@index([userId])
}
```

## 2. 文件存储结构 (CloudFlare R2)

所有文件使用 **CloudFlare R2**（S3 兼容），通过 `@aws-sdk/client-s3` 读写，按用户和日期组织：

```
{bucket}/users/{userId}/entries/{YYYY}/{MM}/
  ├── {YYYY-MM-DD}.md          ← 日记正文 (Markdown)
  └── assets/                ← 图片
      ├── IMG_{timestamp}_{random}.jpg
      └── IMG_{timestamp}_{random}.png
```

> **注意:** Vercel 无持久化文件系统，开发环境 `data/` 仅供本地模拟，生产必须走 R2。

## 4. TypeScript 类型定义

```typescript
// src/types/index.ts

export type Tone = "warm" | "genki" | "minimal" | "literary";

export type ApiProvider = "openai" | "deepseek" | "gemini";

export interface DiaryEntry {
  id: string;
  userId: string;
  date: string;
  title?: string;
  markdown: string;
  tone: Tone;
  tags: string[];
  images: MediaFile[];
  wordCount: number;
  createdAt: string;
  updatedAt: string;
}

export interface DiarySummary {
  id: string;
  date: string;
  title?: string;
  preview: string;
  hasImages: boolean;
  wordCount: number;
  tags: string[];
  createdAt: string;
}

export interface MediaFile {
  url: string;
  path: string;
  type: "image" | "video";
  mime: string;
  size: number;
  width?: number;
  height?: number;
  thumbnail?: string;
}

export interface AIGenerateRequest {
  text: string;
  images: MediaFile[];
  tone: Tone;
  date: string;
}

export interface AIGenerateResponse {
  markdown: string;
  title?: string;
  tags: string[];
}

export interface ApiResponse<T> {
  ok: boolean;
  data?: T;
  error?: string;
}
```
