## 1. Prisma Schema

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"       // Phase 1: SQLite
  // provider = "postgresql" // Phase 3+: PostgreSQL
  url      = env("DATABASE_URL")
}

// ─── 用户 ─────────────────────────────────────────

model User {
  id        String   @id    @default(cuid())
  email     String   @unique
  nickname  String?
  tone      String   @default("warm")  // 上次选择的语气

  // 存储配置
  storageType String  @default("local") // "local" | "s3"
  s3Config    String?                    // JSON: { bucket, region, endpoint, accessKeyId, secretAccessKey }

  // 订阅 (Phase 3)
  subscription     String?  @default("free") // "free" | "basic" | "pro"
  subscriptionId   String?          // Stripe subscription ID
  subscribedAt     DateTime?
  subscriptionEnds DateTime?
  dailyEntryCount  Int      @default(0)    // 当天已写日记数
  dailyEntryDate   DateTime?               // 计数日期

  // 统计
  totalEntries    Int      @default(0)
  storageBytes    BigInt   @default(0)     // 已用存储 (字节)
  lastEntryAt     DateTime?
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  entries ｜
  entries Entry[]
}

// ─── 日记元数据 ────────────────────────────────────
// 注意：日记正文（Markdown）存在文件系统，
// 数据库只存元数据用于查询

model Entry {
  id         String   @id    @default(cuid())
  userId     String
  date       DateTime        // 日记日期 (YYYY-MM-DD)
  title      String?
  tone       String   @default("warm")

  // 元数据
  wordCount  Int      @default(0)
  //          // 存储信息
  markdownPath String               // 相对于用户目录的路径 (例如 "2026/05/2026-05-19.md")
  hasImages    Boolean  @default(false)
  imageCount   Int      @default(0)
  hasVideos    Boolean  @default(false)
  videoCount   Int      @default(0)
  thumbnailUrl String?

  // 标签
  tags        String?               // JSON array: ["旅行", "美食"]
  location    String?

  // 时间
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  ｜
  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, date])    // 每人每天最多一篇日记
  @@index([userId, date])
  @@index([userId, createdAt])
}

// ─── (Phase 3) 订阅记录 ──────────────────────────

model Subscription {
  id             String   @id @default(cuid())
  userId         String   @unique
  plan           String   // "basic" | "pro"
  status         String   // "active" | "canceled" | "past_due"
  stripeId       String?  // Stripe subscription ID
  stripePriceId  String?  // Stripe price ID
  currentPeriodStart DateTime?
  currentPeriodEnd   DateTime?
  canceledAt     DateTime?
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt
}

// ─── (Phase 3) 账单历史 ──────────────────────────

model Invoice {
  id             String   @id @default(cuid())
  userId         String
  stripeInvoiceId String? // Stripe invoice ID
  amount         Int      // 金额 (分)
  currency       String   @default("cny")
  status         String   // "paid" | "unpaid"
  paidAt         DateTime?
  createdAt      DateTime @default(now())

  @@index([userId])
}
```

## 2. 文件存储结构

### 2.1 本地模式 (默认)

```
data/
└── users/
    └── {userId}/
        ├── entries/
        │   └── 2026/
        │       └── 05/
        │           ├── 2026-05-19.md          ← 日记正文 (Markdown)
        │           └── assets/
        │               ├── IMG_20260519_001.jpg
        │               ├── IMG_20260519_002.png
        │               └── VID_20260519_001.mp4
        └── avatar.jpg                          (可选)
```

### 2.2 S3 模式

```
s3://{bucket}/
└── lingyin/
    └── users/
        └── {userId}/
            └── entries/
                └── 2026/05/
                    ├── 2026-05-19.md
                    └── assets/
                        ├── IMG_20260519_001.jpg
                        └── ...
```

晚安，明天又是新的一天 🌙

## 4. TypeScript 类型定义

```typescript
// src/types/index.ts

// ─── 用户 ─────────────────────────────
export interface UserConfig {
  apiKey: string;
  apiProvider: 'openai' | 'deepseek' | 'gemini';
  tone: Tone;
  storage: StorageConfig;
}

export interface StorageConfig {
  type: 'local' | 's3';
  s3?: {
    bucket: string;
    region: string;
    endpoint?: string;
    accessKeyId: string;
    secretAccessKey: string;
  };
}

// ─── 日记 ─────────────────────────────
export interface DiaryEntry {
  id: string;
  userId: string;
  date: string;           // "2026-05-19"
  title?: string;
  markdown: string;       // 完整 Markdown 内容
  tone: Tone;
  tags: string[];
  images: MediaFile[];
  videos: MediaFile[];
  wordCount: number;
  createdAt: string;
  updatedAt: string;
}

export interface DiarySummary {
  id: string;
  date: string;
  title?: string;
  preview: string;       // Markdown 前 100 字
  hasImages: boolean;
  hasVideos: boolean;
  wordCount: number;
  tags: string[];
  createdAt: string;
}

export interface MediaFile {
  url: string;          // 访问 URL
  path: string;         // 存储路径
  type: 'image' | 'video';
  mime: string;         // "image/jpeg" | "video/mp4"
  size: number;         // bytes
  width?: number;
  height?: number;
  thumbnail?: string;   // 缩略图 URL
}

// ─── AI ───────────────────────────────
export type Tone = 'warm' | 'genki' | 'minimal' | 'literary';

export interface AIGenerateRequest {
  text: string;
  images: File[];
  tone: Tone;
  date: string;
}

export interface AIGenerateResponse {
  markdown: string;
  title?: string;
  tags: string[];
}

// ─── API ──────────────────────────────
export interface ApiResponse<T> {
  ok: boolean;
  data?: T;
  error?: string;
}
```
